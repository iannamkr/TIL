input = [1.5, -12.3, 3.2, -5.5, 23.2, 3.2, -1.4, -12.2, 34.2, 5.4, -7.8, 1.1, -4.9]

max_num = 0
p = -1
for i in range(len(input)):
    if input[i] > 0:
        p = i
        break
    elif input[i] > max_num:
        max_num = input[i]
        p = i

if p < 0:
    print("{}, {}, {}".format(max_num, p, p))
else:
    l = 0
    r = 0
    sum_max = 0
    sum_pq = 0
    for q in range(p, len(input)):
        if p < 0:
            if input[q] > 0:
                l = p = q
            else:
                continue

        sum_pq += input[q]

        if sum_pq > sum_max:
            sum_max = sum_pq
            r = q
        elif sum_pq < 0:
            l = p
            p = -1
            sum_pq = 0

print("{}, {}, {}".format(sum_pq, l, r))
